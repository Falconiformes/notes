	\centering
		\begin{tikzpicture}
			\node[fill=darkblue,text=white] {[niveau].[verandering op niveau]};
		\end{tikzpicture}
		\begin{itemize}
		\item niveau
			\begin{itemize}
			\item c $\rightarrow$ gen
			\item p $\rightarrow$ prote\"ine

		\end{itemize}
